def kernel(xa, xb, amp=2.0, ll=0.5):
    D = pairwise_distances(xa, xb)
    return amp**2 * np.exp(-0.5 * 1./ll**2 * D**2 )

def sigmoid(f):
    return 1.0 / (1.0 + np.exp(-f)) + 1e-8
